# get the remote git repo to a local dir (this will create a subdirectory named exercise-roulette-app) 
git clone https://github.com/mekShnayer/exercise-roulette-app.git

# go into the new sub-directory
cd exercise-roulette-app

# create a heroku app, named <APP_NAME>, using mars/create-react-app buildpack
heroku create <APP_NAME> --buildpack mars/create-react-app
#without <>

# you can do whatever changes to the code here
# and then add the changes to git and commit 
git add .
git commit -m "react-create-app on Heroku"

# push the changes to your regular github remote repo
git push 

# push the changes to heroku main branch
git push heroku main

# open your heroku app in the browser 
heroku open
